@startuml
skinparam state {
  BackgroundColor<<Success>> #lightgreen
  BackgroundColor<<Fail>> #pink
}

title Жизненный цикл группы пингвинов (Побег из зоопарка)

[*] --> ВКлетках : <<create>>

state "В клетках" as ВКлетках
state "Сбегают" as Сбегают
state "На корабле" as НаКорабле <<Success>>
state "Побег не удался" as ПобегНеудался <<Fail>>
state c1 <<choice>>
state c2 <<choice>>

' Переходы
ВКлетках --> Сбегают : планАктивирован()

Сбегают --> c1 : выполнитьПлан()

' Точка выбора 1: проверка взлома и усыпления
c1 --> НаКорабле : [взломУспешен && охранаУсыплена]
c1 --> ПобегНеудался : [взломПровален || охранаОбнаружена]

' Переход к захвату корабля
НаКорабле --> c2 : захватитьСудно()

' Точка выбора 2: проверка нейтрализации капитана
c2 --> [*] : [капитанНейтрализован] \n Успешный побег!
c2 --> ПобегНеудался : [капитанСопротивляется]

' Финал при провале
ПобегНеудался --> [*] : <<destroy>>

note left of c1
  **Точка выбора (Choice 1)**
  Проверяет:
  - Взлом замка (Ковальски)
  - Усыпление охраны (Рико)
  - Отвлечение внимания (Младший)
end note

note right of c2
  **Точка выбора (Choice 2)**
  Проверяет:
  - Нейтрализация капитана
  Шкипер руководит захватом
end note

note bottom of НаКорабле
  Состояние достигнуто,
  корабль захвачен успешно
end note

@enduml
